## 1. 部署概述

### 1.1 部署目标
- **高可用性**: 确保系统7x24小时稳定运行
- **高性能**: 支持1000+并发用户访问
- **可扩展性**: 支持水平扩展和垂直扩展
- **安全性**: 确保数据安全和系统安全
- **易维护性**: 便于运维管理和故障排查

### 1.2 部署架构
```
                    ┌─────────────────┐
                    │   Load Balancer │
                    │    (Nginx)      │
                    └─────────┬───────┘
                              │
                    ┌─────────┴───────┐
                    │                 │
            ┌───────▼────────┐ ┌──────▼────────┐
            │  Web Server 1  │ │  Web Server 2 │
            │   (Gunicorn)   │ │   (Gunicorn)  │
            └───────┬────────┘ └──────┬────────┘
                    │                 │
                    └─────────┬───────┘
                              │
                    ┌─────────▼───────┐
                    │   Database      │
                    │    (MySQL)      │
                    └─────────────────┘
                              │
                    ┌─────────▼───────┐
                    │     Cache       │
                    │    (Redis)      │
                    └─────────────────┘
```

### 1.3 技术栈
- **操作系统**: Ubuntu 22.04 LTS
- **Web服务器**: Nginx 1.22+
- **应用服务器**: Gunicorn 20.1+
- **数据库**: MySQL 8.0+
- **缓存**: Redis 7.0+
- **任务队列**: Celery 5.3+
- **监控**: Prometheus + Grafana
- **日志**: ELK Stack (Elasticsearch + Logstash + Kibana)

#### 3.3.2 Nginx配置
```nginx
# /etc/nginx/sites-available/familytree
upstream familytree_backend {
    server 127.0.0.1:8000;
    # 如果有多个后端服务器
    # server 10.0.0.101:8000;
    # server 10.0.0.102:8000;
}

server {
    listen 80;
    server_name your-domain.com www.your-domain.com;
    
    # 重定向到HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-domain.com www.your-domain.com;
    
    # SSL证书配置
    ssl_certificate /etc/ssl/certs/your-domain.crt;
    ssl_certificate_key /etc/ssl/private/your-domain.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512:ECDHE-RSA-AES256-GCM-SHA384:DHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    
    # 安全头
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    
    # 客户端最大上传大小
    client_max_body_size 50M;
    
    # 静态文件
    location /static/ {
        alias /var/www/familytree/static/;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
    
    location /media/ {
        alias /var/www/familytree/media/;
        expires 7d;
        add_header Cache-Control "public";
    }
    
    # API请求
    location /api/ {
        proxy_pass http://familytree_backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_connect_timeout 30s;
        proxy_send_timeout 30s;
        proxy_read_timeout 30s;
    }
    
    # 前端应用
    location / {
        try_files $uri $uri/ /index.html;
        root /var/www/familytree/frontend/dist;
        index index.html;
        
        # 缓存策略
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
    }
    
    # 健康检查
    location /health {
        access_log off;
        return 200 "healthy\n";
        add_header Content-Type text/plain;
    }
}
```

#### 6.2.2 自定义监控指标
```python
# monitoring/metrics.py
from prometheus_client import Counter, Histogram, Gauge
from django.conf import settings

# 用户注册计数器
user_registration_counter = Counter(
    'familytree_user_registrations_total',
    'Total number of user registrations'
)

# 族谱创建计数器
family_tree_creation_counter = Counter(
    'familytree_family_trees_created_total',
    'Total number of family trees created'
)

    'familytree_request_duration_seconds',
    'Time spent processing requests',
```

```typescript
export class GraphOptimizer {
  /**
   * 优化大型图形数据
   */
  static optimizeLargeGraph(nodes: any[], edges: any[]) {
    // 节点数量超过阈值时进行优化
    if (nodes.length > 1000) {
      return this.applyLevelOfDetail(nodes, edges);
    }
    return { nodes, edges };
  }
  
  /**
   * 应用细节层次优化
   */
  private static applyLevelOfDetail(nodes: any[], edges: any[]) {
    // 根据缩放级别显示不同详细程度的节点
    const optimizedNodes = nodes.map(node => ({
      ...node,
      // 简化节点样式
      style: this.getSimplifiedNodeStyle(node)
    }));
    
    // 过滤不重要的边
    const optimizedEdges = edges.filter(edge => 
      this.isImportantEdge(edge)
    );
    
    return {
      nodes: optimizedNodes,
      edges: optimizedEdges
    };
  }
  
  /**
   * 虚拟滚动优化
   */
  static createVirtualizedList(items: any[], viewportHeight: number) {

    const visibleCount = Math.ceil(viewportHeight / itemHeight);

      visibleCount,
```

```python
from django.utils.html import escape

class SecurityValidator:
    """安全验证器"""
    
    @staticmethod
    def validate_filename(filename):
        """验证文件名安全性"""
        # 检查文件名长度
        if len(filename) > 255:
            raise ValidationError("文件名过长")
        
        # 检查危险字符
        dangerous_chars = ['..', '/', '\\', '<', '>', ':', '"', '|', '?', '*']
        for char in dangerous_chars:
            if char in filename:
                raise ValidationError("文件名包含非法字符")
        
        return True
    
    @staticmethod
    def sanitize_html_input(text):
        """清理HTML输入"""
        # 转义HTML特殊字符
        return escape(text)
    
    @staticmethod
    def validate_sql_injection(text):
        """检查SQL注入"""
        sql_keywords = [
            'SELECT', 'INSERT', 'UPDATE', 'DELETE', 'DROP', 'CREATE',
            'ALTER', 'EXEC', 'UNION', 'SCRIPT', 'JAVASCRIPT'
        ]
        
        text_upper = text.upper()
        for keyword in sql_keywords:
            if keyword in text_upper:
                raise ValidationError("输入包含非法内容")
        
        return True
```
