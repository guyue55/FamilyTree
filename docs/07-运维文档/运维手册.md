## 1. 系统概述

### 1.1 系统架构
族谱系统采用前后端分离架构：
- **前端**: Vue.js 3 + Element Plus
- **后端**: Django 4.2 + Django-Ninja
- **数据库**: MySQL 8.0
- **缓存**: Redis 7
- **任务队列**: Celery
- **Web服务器**: Nginx + Gunicorn
- **监控**: Prometheus + Grafana

### 1.2 服务器配置

#### 1.2.1 生产环境

```nginx
server {
    listen 80;
    server_name your-domain.com;
    
    # 重定向到HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-domain.com;
    
    # SSL配置
    ssl_certificate /etc/letsencrypt/live/your-domain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/your-domain.com/privkey.pem;
    
    # 安全头
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    
    # 静态文件
    location /static/ {
        alias /opt/familytree/static/;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
    
    # 媒体文件
    location /media/ {
        alias /opt/familytree/media/;
        expires 7d;
    }
    
    # API请求
    location /api/ {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
    
    # 前端应用
    location / {
        try_files $uri $uri/ /index.html;
        root /opt/familytree/frontend/dist;
    }
}
```

## 3. 服务管理

### 3.1 Systemd服务配置

#### 3.1.1 Django服务配置
```ini
# /etc/systemd/system/familytree-django.service
[Unit]
Description=FamilyTree Django Application
After=network.target mysql.service redis.service

[Service]
Type=exec
User=familytree
Group=familytree
WorkingDirectory=/opt/familytree/backend
Environment=PATH=/opt/familytree/venv/bin
ExecStart=/opt/familytree/venv/bin/gunicorn \
    --workers 4 \
    --bind 127.0.0.1:8000 \
    --timeout 120 \
    --max-requests 1000 \
    --max-requests-jitter 100 \
    --preload \
    familytree.wsgi:application
ExecReload=/bin/kill -s HUP $MAINPID
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target
```

#### 3.1.2 Celery服务配置
```ini
# /etc/systemd/system/familytree-celery.service
[Unit]

User=familytree
Group=familytree
WorkingDirectory=/opt/familytree/backend
Environment=PATH=/opt/familytree/venv/bin

    --loglevel=info \
    --concurrency=4 \
    --max-tasks-per-child=1000
ExecReload=/bin/kill -s HUP $MAINPID
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target
```

#### 3.1.3 Celery Beat服务配置

## 9. 备份恢复

### 9.1 数据库备份

#### 9.1.1 自动备份脚本
```bash
#!/bin/bash
# /opt/familytree/scripts/backup_database.sh

BACKUP_DIR="/opt/familytree/backups"
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME="familytree"
DB_USER="familytree_user"

# 创建备份目录
mkdir -p $BACKUP_DIR

# 数据库备份
echo "开始数据库备份..."
mysqldump -h localhost -u $DB_USER -p$DB_PASSWORD $DB_NAME | gzip > $BACKUP_DIR/db_backup_$DATE.sql.gz

# 删除7天前的备份
find $BACKUP_DIR -name "db_backup_*.sql.gz" -mtime +7 -delete

echo "数据库备份完成: $BACKUP_DIR/db_backup_$DATE.sql.gz"
```

#### 9.1.2 数据库恢复
```bash
#!/bin/bash
# /opt/familytree/scripts/restore_database.sh

BACKUP_FILE=$1
DB_NAME="familytree"
DB_USER="familytree_user"

if [ -z "$BACKUP_FILE" ]; then
    echo "用法: $0 <backup_file>"
    exit 1
fi

echo "停止应用服务..."
sudo systemctl stop familytree-django
sudo systemctl stop familytree-celery

echo "恢复数据库..."
gunzip -c $BACKUP_FILE | mysql -h localhost -u $DB_USER -p$DB_PASSWORD $DB_NAME

echo "启动应用服务..."
sudo systemctl start familytree-django
sudo systemctl start familytree-celery

echo "数据库恢复完成"
```

```bash
#!/bin/bash
# /opt/familytree/scripts/system_monitor.sh

THRESHOLD_CPU=80

# 检查CPU使用率
cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1 | cut -d',' -f1)
if (( $(echo "$cpu_usage > $THRESHOLD_CPU" | bc -l) )); then

    echo "警告: 内存使用率过高 ($mem_usage%)"
fi

# 检查磁盘使用率
disk_usage=$(df -h / | awk 'NR==2{print $5}' | cut -d'%' -f1)
if [ "$disk_usage" -gt "$THRESHOLD_DISK" ]; then
    echo "警告: 磁盘使用率过高 ($disk_usage%)"
fi

# 检查服务状态
services=("familytree-django" "familytree-celery" "nginx" "mysql" "redis-server")
for service in "${services[@]}"; do
    if ! systemctl is-active --quiet $service; then
```

```bash
#!/bin/bash
# /opt/familytree/scripts/performance_monitor.sh

LOG_FILE="/opt/familytree/logs/performance.log"
DATE=$(date '+%Y-%m-%d %H:%M:%S')

# 获取系统负载

mem_total=$(free -m | awk 'NR==2{print $2}')

# 获取Redis内存使用
redis_memory=$(redis-cli info memory | grep used_memory_human | cut -d: -f2 | tr -d '\r')

# 记录到日志
echo "$DATE,Load:$load_avg,Memory:$mem_percent%,DB_Conn:$db_connections,Redis:$redis_memory" >> $LOG_FILE
```

#### B.2 Nginx配置模板
```nginx
# 基础配置模板
server {
    listen 80;
    server_name _;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-domain.com;
    
    ssl_certificate /path/to/cert.pem;
    ssl_certificate_key /path/to/key.pem;
    
    location /static/ {
        alias /opt/familytree/static/;
        expires 30d;
    }
    
    location /media/ {
        alias /opt/familytree/media/;
        expires 7d;
    }
    
    location /api/ {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
    
    location / {
        try_files $uri $uri/ /index.html;
        root /opt/familytree/frontend/dist;
    }
}
```
