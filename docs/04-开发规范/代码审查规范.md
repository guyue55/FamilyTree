import type { UploadProps } from 'element-plus'

interface Props {
  modelValue?: string
  size?: 'small' | 'default' | 'large'
}

interface Emits {
  (e: 'update:modelValue', value: string): void
  (e: 'change', value: string): void
}

  size: 'default'
})

const beforeUpload: UploadProps['beforeUpload'] = (file) => {
  const isImage = file.type.startsWith('image/')
  const isLt2M = file.size / 1024 / 1024 < 2

  if (!isImage) {
    ElMessage.error('只能上传图片文件!')
    return false
  }
  if (!isLt2M) {
    ElMessage.error('图片大小不能超过 2MB!')
    return false
  }
  return true

const handleSuccess = (response: any) => {
  if (response.code === 201) {
    const newUrl = response.data.upload_url
    emit('update:modelValue', newUrl)
    emit('change', newUrl)
    ElMessage.success('头像上传成功')
  } else {
    ElMessage.error(response.message || '上传失败')
  }
}

// 上传失败处理
const handleError = () => {
  ElMessage.error('上传失败，请重试')
}

```typescript
// stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { User, LoginForm, RegisterForm } from '@/types/user'
import { userApi } from '@/api/user'

export const useUserStore = defineStore('user', () => {
  // 状态
  const user = ref<User | null>(null)
  const token = ref<string>('')
  const loading = ref(false)

  // 计算属性
  const isLoggedIn = computed(() => !!token.value)
  const userName = computed(() => user.value?.real_name || user.value?.username || '')

  // 操作
  const login = async (form: LoginForm) => {
    loading.value = true
    try {
      const response = await userApi.login(form)
      token.value = response.data.access_token
      user.value = response.data.user
      
      // 保存到本地存储
      localStorage.setItem('token', token.value)
      localStorage.setItem('user', JSON.stringify(user.value))
      
      return response
    } catch (error) {
      throw error
    } finally {
      loading.value = false
    }
  }

  const logout = () => {
    user.value = null
    token.value = ''
    localStorage.removeItem('token')
    localStorage.removeItem('user')
  }

  const updateProfile = async (data: Partial<User>) => {
    loading.value = true
    try {
      const response = await userApi.updateProfile(data)
      user.value = { ...user.value!, ...response.data }
      localStorage.setItem('user', JSON.stringify(user.value))
      return response
    } catch (error) {
      throw error
    } finally {
      loading.value = false
    }
  }

  // 初始化
  const initializeAuth = () => {
    const savedToken = localStorage.getItem('token')
    const savedUser = localStorage.getItem('user')
    
    if (savedToken && savedUser) {
      token.value = savedToken
      user.value = JSON.parse(savedUser)
    }
  }

  return {
    // 状态
    user: readonly(user),
    token: readonly(token),
    loading: readonly(loading),
    
    // 计算属性
    isLoggedIn,
    userName,
    
    // 操作
    login,
    logout,
    updateProfile,
    initializeAuth
  }
})

        return cached_data

            "message": "success",

            }

from django.db import models
from django.contrib.auth import get_user_model
from django.core.validators import MinLengthValidator, MaxLengthValidator
from django.utils import timezone
from django.db.models import Q
import uuid

User = get_user_model()

class Family(models.Model):
    """家族模型"""
    
    # 基础信息
    id = models.AutoField(primary_key=True)
    uuid = models.UUIDField(default=uuid.uuid4, unique=True, editable=False)
    name = models.CharField(
        max_length=100,
        validators=[MinLengthValidator(2)],
        verbose_name="家族名称",
        help_text="家族的正式名称"
    )
    description = models.TextField(
        max_length=1000,
        blank=True,
        verbose_name="家族描述",
        help_text="家族的详细描述"
    )
    
    # 家族详情
    origin_place = models.CharField(
        max_length=200,
        blank=True,
        verbose_name="发源地",
        help_text="家族的发源地"
    )
    founder_name = models.CharField(
        max_length=50,
        blank=True,
        verbose_name="始祖姓名",
        help_text="家族始祖的姓名"

    founded_year = models.PositiveIntegerField(
        null=True,
        blank=True,
        verbose_name="创建年份",
        help_text="家族创建的年份"
    )
    family_motto = models.CharField(

    )
    family_rules = models.TextField(
        blank=True,
        verbose_name="家族规则",

    )

    cover_image = models.ImageField(
        upload_to='family_covers/%Y/%m/',
        blank=True,
        null=True,
        verbose_name="封面图片"
    )
    
    # 设置
    is_public = models.BooleanField(
        default=True,
        verbose_name="是否公开",
        help_text="是否允许其他用户查看"
    )
    allow_join_request = models.BooleanField(
        default=True,
        verbose_name="允许加入申请",
        help_text="是否允许用户申请加入"
    )
    
    # 关联
    creator_id = models.BigIntegerField(
        verbose_name="创建者ID",
        help_text="逻辑关联 users.id"
    )
    members = models.ManyToManyField(
        User,
        through='FamilyMember',
        related_name='families',
        verbose_name="成员"
    )
    
    # 时间戳
    created_at = models.DateTimeField(auto_now_add=True, verbose_name="创建时间")
    updated_at = models.DateTimeField(auto_now=True, verbose_name="更新时间")
    
    class Meta:
        db_table = 'families'
        verbose_name = '家族'
        verbose_name_plural = '家族'
        ordering = ['-created_at']
        indexes = [
            models.Index(fields=['name']),
            models.Index(fields=['is_public']),
            models.Index(fields=['creator_id']),
            models.Index(fields=['created_at']),
        ]
    
    def __str__(self):
        return self.name
    
    @property
    def creator(self):
        """获取创建者"""
        from django.contrib.auth.models import User
        return User.objects.get(id=self.creator_id)
    
    @property
    def member_count(self):
        """成员数量"""
        return self.members.count()
    
    def get_admin_members(self):
        """获取管理员成员"""
        return self.familymember_set.filter(role='admin').select_related('user')
    
    def can_user_view(self, user):
        """检查用户是否可以查看家族"""
        if self.is_public:
            return True
        if user.is_authenticated:
            return self.members.filter(id=user.id).exists()
        return False
    
    def can_user_edit(self, user):
        """检查用户是否可以编辑家族"""
        if not user.is_authenticated:
            return False
        if user.id == self.creator_id:
            return True
        return self.familymember_set.filter(
            user=user,
            role__in=['admin', 'moderator']
        ).exists()

def calculate_relationship_degree(person1_id: int, person2_id: int) -> int:
    """
    计算两个人之间的亲属关系度数
    
    Args:
        person1_id: 第一个人的ID
        person2_id: 第二个人的ID
    
    Returns:
        int: 关系度数，0表示同一人，-1表示无关系
    
    Examples:
        >>> calculate_relationship_degree(1, 2)  # 父子关系
        1
        >>> calculate_relationship_degree(1, 3)  # 祖孙关系
        2
    
    Note:
        使用广度优先搜索算法查找最短路径
    """
    if person1_id == person2_id:
        return 0
    
    # 使用BFS查找最短路径
    visited = set()
    queue = [(person1_id, 0)]
    
    while queue:
        current_id, degree = queue.pop(0)
        
        if current_id in visited:
            continue
            
        visited.add(current_id)
        
        # 查找直接关系
        relatives = get_direct_relatives(current_id)
        
        for relative_id in relatives:
            if relative_id == person2_id:
                return degree + 1
            
            if relative_id not in visited:
                queue.append((relative_id, degree + 1))
    
    return -1  # 无关系
```
