## 2. 数据库命名规范

### 2.1 数据库命名
```sql
-- 开发环境
familytree_dev

-- 测试环境
familytree_test

-- 生产环境
familytree_prod
```

### 2.2 表命名规范
- 使用小写字母和下划线
- 表名使用复数形式
- 避免使用保留字
- 表名要有意义且简洁

```sql
-- 正确示例
users                    -- 用户表
families                 -- 家族表
family_members          -- 家族成员关联表
member_relationships    -- 成员关系表
media_files            -- 媒体文件表

-- 错误示例
User                   -- 大写字母
user                   -- 单数形式
t_user                 -- 无意义前缀
```

### 2.3 字段命名规范
- 使用小写字母和下划线
- 字段名要有意义
- 布尔字段使用 `is_` 前缀
- 时间字段使用 `_at` 后缀
- 外键字段使用 `_id` 后缀

```sql
-- 正确示例
id                     -- 主键
user_id               -- 外键
username              -- 用户名
email                 -- 邮箱
is_active             -- 布尔字段
is_deleted            -- 软删除标记
created_at            -- 创建时间
updated_at            -- 更新时间
birth_date            -- 出生日期

-- 错误示例
ID                    -- 大写
userId                -- 驼峰命名
active                -- 布尔字段缺少is_前缀
create_time           -- 时间字段缺少_at后缀
```

### 2.4 索引命名规范
```sql

pk_users

uk_users_email
uk_users_phone

-- 复合索引
```

### 2.5 约束命名规范
```sql

fk_member_relationships_parent_id
```

## 4. 表结构设计规范

### 4.1 基础表结构
所有业务表都应继承基础字段：

```sql
CREATE TABLE base_model (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_deleted BOOLEAN DEFAULT FALSE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

### 4.2 用户表设计
```sql
CREATE TABLE users (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    phone VARCHAR(20) UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    real_name VARCHAR(50),
    avatar_url VARCHAR(500),
    gender VARCHAR(10) CHECK (gender IN ('male', 'female', 'unknown')),
    birth_date DATE,
    is_active BOOLEAN DEFAULT TRUE,
    is_staff BOOLEAN DEFAULT FALSE,
    is_superuser BOOLEAN DEFAULT FALSE,
    last_login TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_deleted BOOLEAN DEFAULT FALSE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 索引
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_phone ON users(phone);
CREATE INDEX idx_users_created_at ON users(created_at);
CREATE INDEX idx_users_is_active ON users(is_active);
```

### 4.3 家族表设计
```sql
CREATE TABLE families (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    origin_place VARCHAR(200),
    founder_name VARCHAR(50),
    founded_year INT,
    family_motto TEXT,
    family_rules TEXT,
    cover_image_url VARCHAR(500),
    is_public BOOLEAN DEFAULT TRUE,
    creator_id BIGINT NOT NULL COMMENT '创建者ID，逻辑关联users.id',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_deleted BOOLEAN DEFAULT FALSE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 索引
CREATE INDEX idx_families_creator_id ON families(creator_id);
CREATE INDEX idx_families_name ON families(name);
CREATE INDEX idx_families_is_public ON families(is_public);
CREATE INDEX idx_families_created_at ON families(created_at);
```

### 4.4 家族成员关联表
```sql
CREATE TABLE family_members (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    family_id BIGINT NOT NULL COMMENT '家族ID，逻辑关联families.id',
    user_id BIGINT NOT NULL COMMENT '用户ID，逻辑关联users.id',
    role VARCHAR(30) DEFAULT 'member' CHECK (role IN ('admin', 'editor', 'member')),
    join_date DATE DEFAULT (CURRENT_DATE),
    status VARCHAR(20) DEFAULT 'active' CHECK (status IN ('active', 'pending', 'inactive')),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_deleted BOOLEAN DEFAULT FALSE,
    
    CONSTRAINT uk_family_members_family_user UNIQUE (family_id, user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 索引
CREATE INDEX idx_family_members_family_id ON family_members(family_id);
CREATE INDEX idx_family_members_user_id ON family_members(user_id);
CREATE INDEX idx_family_members_role ON family_members(role);
CREATE INDEX idx_family_members_status ON family_members(status);
```

### 4.5 成员关系表
```sql
CREATE TABLE member_relationships (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    family_id BIGINT NOT NULL COMMENT '家族ID，逻辑关联families.id',
    parent_id BIGINT NOT NULL COMMENT '父级成员ID，逻辑关联users.id',
    child_id BIGINT NOT NULL COMMENT '子级成员ID，逻辑关联users.id',
    relationship_type VARCHAR(30) NOT NULL CHECK (
        relationship_type IN ('father', 'mother', 'spouse', 'sibling')
    ),
    generation_diff INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_deleted BOOLEAN DEFAULT FALSE,
    
    CONSTRAINT ck_member_relationships_not_self CHECK (parent_id != child_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 索引
CREATE INDEX idx_member_relationships_family_id ON member_relationships(family_id);
CREATE INDEX idx_member_relationships_parent_id ON member_relationships(parent_id);
CREATE INDEX idx_member_relationships_child_id ON member_relationships(child_id);
CREATE INDEX idx_member_relationships_type ON member_relationships(relationship_type);
CREATE UNIQUE INDEX uk_member_relationships_unique ON member_relationships(family_id, parent_id, child_id, relationship_type);
```

## 5. 索引设计规范

### 5.1 索引类型选择
```sql
-- B-tree索引（默认）：适用于等值查询和范围查询
CREATE INDEX idx_users_created_at ON users(created_at);

-- 唯一索引：确保数据唯一性
CREATE UNIQUE INDEX uk_users_email ON users(email);

-- 复合索引：多字段查询优化
CREATE INDEX idx_family_members_family_user ON family_members(family_id, user_id);

-- 前缀索引：节省空间
CREATE INDEX idx_users_email_prefix ON users(email(20));

-- 全文索引：全文搜索
CREATE FULLTEXT INDEX idx_families_search ON families(name, description);
```

### 5.2 索引设计原则
1. **主键自动创建索引**
2. **关联字段创建索引** - 用于逻辑关联的字段
3. **频繁查询字段创建索引**
4. **复合查询创建复合索引**
5. **避免过多索引影响写入性能**

### 5.3 索引监控
```sql
-- 查看索引使用情况
SELECT 
    TABLE_SCHEMA,
    TABLE_NAME,
    INDEX_NAME,
    CARDINALITY,
    INDEX_TYPE
FROM INFORMATION_SCHEMA.STATISTICS
WHERE TABLE_SCHEMA = 'familytree_prod'
ORDER BY CARDINALITY DESC;

-- 查看表大小和索引大小
SELECT 
    TABLE_NAME,
    ROUND(((DATA_LENGTH + INDEX_LENGTH) / 1024 / 1024), 2) AS 'DB Size in MB',
    ROUND((DATA_LENGTH / 1024 / 1024), 2) AS 'Data Size in MB',
    ROUND((INDEX_LENGTH / 1024 / 1024), 2) AS 'Index Size in MB'
FROM INFORMATION_SCHEMA.TABLES
WHERE TABLE_SCHEMA = 'familytree_prod'
ORDER BY (DATA_LENGTH + INDEX_LENGTH) DESC;
```

## 7. 数据库安全规范

### 7.1 用户权限管理
```sql
-- 创建应用用户
CREATE USER 'familytree_app'@'%' IDENTIFIED BY 'strong_password';

-- 授予必要权限
GRANT SELECT, INSERT, UPDATE, DELETE ON familytree_prod.* TO 'familytree_app'@'%';

-- 只读用户
CREATE USER 'familytree_readonly'@'%' IDENTIFIED BY 'readonly_password';
GRANT SELECT ON familytree_prod.* TO 'familytree_readonly'@'%';
FLUSH PRIVILEGES;
```

### 7.2 敏感数据处理
```sql
-- 密码字段加密存储
password_hash VARCHAR(255) NOT NULL  -- 存储哈希值，不存储明文

-- 敏感信息字段
phone VARCHAR(20)  -- 可考虑加密存储
id_card VARCHAR(50)  -- 身份证号加密存储
```

### 7.3 数据访问控制
```sql
-- 使用视图控制数据访问
CREATE VIEW user_families AS
SELECT f.* FROM families f
JOIN family_members fm ON f.id = fm.family_id
WHERE fm.user_id = @current_user_id;

-- 使用存储过程控制数据操作
DELIMITER //
CREATE PROCEDURE GetUserFamilies(IN user_id BIGINT)
BEGIN
    SELECT f.* FROM families f
    JOIN family_members fm ON f.id = fm.family_id
    WHERE fm.user_id = user_id;
END //
DELIMITER ;
```

## 8. 性能优化规范

### 8.1 查询优化
```sql
-- 使用EXPLAIN分析查询计划
EXPLAIN FORMAT=JSON 
SELECT * FROM users WHERE email = 'test@example.com';

-- 避免SELECT *
SELECT id, username, email FROM users WHERE is_active = TRUE;

-- 使用LIMIT限制结果集
SELECT * FROM families ORDER BY created_at DESC LIMIT 20;
```

### 8.2 分页查询优化
```sql
-- 使用游标分页（推荐）
SELECT * FROM families 
WHERE id > 1000 
ORDER BY id 
LIMIT 20;

-- 避免OFFSET分页（大偏移量性能差）
SELECT * FROM families 
ORDER BY created_at DESC 
LIMIT 20 OFFSET 10000;  -- 不推荐
```

### 8.3 批量操作优化
```sql
-- 批量插入
INSERT INTO family_members (family_id, user_id, role) VALUES
    (1, 101, 'member'),
    (1, 102, 'member'),
    (1, 103, 'editor');

-- 批量更新
UPDATE users 
SET updated_at = CURRENT_TIMESTAMP 
WHERE id IN (1, 2, 3, 4, 5);
```

## 9. 备份与恢复规范

### 9.1 备份策略
```bash
# 全量备份
mysqldump -h localhost -u root -p familytree_prod > backup_$(date +%Y%m%d_%H%M%S).sql

# 压缩备份
mysqldump -h localhost -u root -p familytree_prod | gzip > backup_$(date +%Y%m%d_%H%M%S).sql.gz

# 仅结构备份
mysqldump -h localhost -u root -p --no-data familytree_prod > schema_$(date +%Y%m%d_%H%M%S).sql

# 仅数据备份
mysqldump -h localhost -u root -p --no-create-info familytree_prod > data_$(date +%Y%m%d_%H%M%S).sql
```

### 9.2 恢复策略
```bash
# 从SQL文件恢复
mysql -h localhost -u root -p familytree_prod < backup_20250127_120000.sql

# 从压缩文件恢复
gunzip < backup_20250127_120000.sql.gz | mysql -h localhost -u root -p familytree_prod
```

### 9.3 备份计划
- **每日备份**: 自动化全量备份
- **每周备份**: 长期存储备份
- **实时备份**: binlog日志备份（主从复制）
