## 5. 状态管理规范

### 5.1 Store结构
```typescript
// stores/family.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { FamilyTree, FamilyMember } from '@/types/family'
import { familyApi } from '@/api/family'

export const useFamilyStore = defineStore('family', () => {
  // 状态定义
  const familyTrees = ref<FamilyTree[]>([])
  const currentTree = ref<FamilyTree | null>(null)
  const currentMembers = ref<FamilyMember[]>([])
  const loading = ref(false)
  const error = ref<string | null>(null)
  
  // 计算属性
  const treeCount = computed(() => familyTrees.value.length)
  const memberCount = computed(() => currentMembers.value.length)
  const hasCurrentTree = computed(() => currentTree.value !== null)
  
  // 操作方法
  const fetchFamilyTrees = async (): Promise<void> => {
    try {
      loading.value = true
      error.value = null
      const response = await familyApi.getFamilyTrees()
      familyTrees.value = response.data
    } catch (err) {
      error.value = '获取族谱列表失败'
      throw err
    } finally {
      loading.value = false
    }
  }
  
  const setCurrentTree = (tree: FamilyTree): void => {
    currentTree.value = tree
  }
  
  const addFamilyTree = async (treeData: Partial<FamilyTree>): Promise<FamilyTree> => {
    try {
      loading.value = true
      const response = await familyApi.createFamilyTree(treeData)
      familyTrees.value.push(response.data)
      return response.data
    } catch (err) {
      error.value = '创建族谱失败'
      throw err
    } finally {
      loading.value = false
    }
  }
  
  const updateFamilyTree = async (id: number, treeData: Partial<FamilyTree>): Promise<void> => {
    try {
      loading.value = true
      const response = await familyApi.updateFamilyTree(id, treeData)
      const index = familyTrees.value.findIndex(tree => tree.id === id)
      if (index !== -1) {
        familyTrees.value[index] = response.data
      }
      if (currentTree.value?.id === id) {
        currentTree.value = response.data
      }
    } catch (err) {
      error.value = '更新族谱失败'
      throw err
    } finally {
      loading.value = false
    }
  }
  
  const deleteFamilyTree = async (id: number): Promise<void> => {
    try {
      loading.value = true
      await familyApi.deleteFamilyTree(id)
      familyTrees.value = familyTrees.value.filter(tree => tree.id !== id)
      if (currentTree.value?.id === id) {
        currentTree.value = null
      }
    } catch (err) {
      error.value = '删除族谱失败'
      throw err
    } finally {
      loading.value = false
    }
  }
  
  // 重置状态
  const resetState = (): void => {
    familyTrees.value = []
    currentTree.value = null
    currentMembers.value = []
    loading.value = false
    error.value = null
  }
  
  return {
    // 状态
    familyTrees,
    currentTree,
    currentMembers,
    loading,
    error,
    
    // 计算属性
    treeCount,
    memberCount,
    hasCurrentTree,
    
    // 操作方法
    fetchFamilyTrees,
    setCurrentTree,
    addFamilyTree,
    updateFamilyTree,
    deleteFamilyTree,
    resetState
  }
})
```

// router/guards.ts
import type { Router } from 'vue-router'
import { useAuthStore } from '@/stores/auth'

export function setupRouterGuards(router: Router): void {
  // 全局前置守卫
  router.beforeEach(async (to, from, next) => {
    const authStore = useAuthStore()
    
    // 设置页面标题
    if (to.meta.title) {
      document.title = `${to.meta.title} - 族谱系统`
    }
    
    // 检查认证状态
    if (to.meta.requiresAuth && !authStore.isAuthenticated) {
      next({ name: 'Login', query: { redirect: to.fullPath } })
      return
    }
    
    // 已登录用户访问认证页面，重定向到首页
    if (authStore.isAuthenticated && to.path.startsWith('/auth')) {
      next({ name: 'Home' })
      return
    }
    
    next()
  })
  
  // 全局后置钩子
  router.afterEach((to) => {
    // 页面访问统计
    console.log(`访问页面: ${to.path}`)
  })
}

## 7. API调用规范

### 7.1 API封装
```typescript
// api/index.ts
import axios from 'axios'
import type { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { ElMessage } from 'element-plus'
import { useAuthStore } from '@/stores/auth'

// 创建axios实例
const request: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
request.interceptors.request.use(
  (config: AxiosRequestConfig) => {
    const authStore = useAuthStore()
    
    // 添加认证token
    if (authStore.token) {
      config.headers = {
        ...config.headers,
        Authorization: `Bearer ${authStore.token}`
      }
    }
    
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
request.interceptors.response.use(
  (response: AxiosResponse) => {
    const { code, message, data } = response.data
    
    if (code === 200) {
      return { data, message }
    } else {
      ElMessage.error(message || '请求失败')
      return Promise.reject(new Error(message))
    }
  },
  (error) => {
    const { response } = error
    
    if (response?.status === 401) {
      const authStore = useAuthStore()
      authStore.logout()
      ElMessage.error('登录已过期，请重新登录')
    } else if (response?.status === 403) {
      ElMessage.error('没有权限访问')
    } else if (response?.status >= 500) {
      ElMessage.error('服务器错误，请稍后重试')
    } else {
      ElMessage.error(response?.data?.message || '网络错误')
    }
    
    return Promise.reject(error)
  }
)

export default request
```

### 12.2 缓存策略
```typescript
// composables/useCache.ts
import { ref, computed } from 'vue'

const cache = new Map()

export function useCache<T>(key: string, fetcher: () => Promise<T>) {
  const data = ref<T | null>(null)
  const loading = ref(false)
  const error = ref<Error | null>(null)
  
  const fetch = async () => {
    if (cache.has(key)) {
      data.value = cache.get(key)
      return
    }
    
    try {
      loading.value = true
      error.value = null
      const result = await fetcher()
      data.value = result
      cache.set(key, result)
    } catch (err) {
      error.value = err as Error
    } finally {
      loading.value = false
    }
  }
  
  return {
    data,
    loading,
    error,
    fetch
  }
}
```

## 13. 版本控制规范

### 13.1 Git提交规范

#### 13.1.1 提交格式
Git commit 要清晰明了的说明本次提交的目的。

**格式**: `[type](scope): subject`

**示例**:
- `[feature](family): add quality field`
- `[bugfix](auth): fix user login validation`

#### 13.1.2 提交类型（Type）
[强制] type 用于说明 git commit 的类别，只允许使用下面的标识：

- **[feature]**: 新功能（feature）
- **[bugfix]**: 修复bug，可以是QA发现的BUG，也可以是研发自己发现的BUG
  - [bugfix]: 产生diff并自动修复此问题。适合于一次提交直接修复问题
  - to: 只产生diff不自动修复此问题。适合于多次提交。最终修复问题提交时使用fix
- **[docs]**: 文档（documentation）
- **[format]**: 格式（不影响代码运行的变动）
- **[refactor]**: 重构（即不是新增功能，也不是修改bug的代码变动）
- **[performance]**: 优化相关，比如提升性能、体验
- **[test]**: 增加测试
- **[chore]**: 构建过程或辅助工具的变动
- **[migrate]**: 代码迁移

#### 13.1.3 作用域（Scope）
scope 用于说明 commit 影响的范围，比如组件、页面、工具等等，视项目不同而不同。

#### 13.1.4 主题（Subject）
[强制] subject是commit目的的简短描述，不超过50个字符：
- 优先使用英文，也可以使用中文
- 结尾不需要标点符号

#### 13.1.5 提交示例
```
[feature](family): 添加族谱可视化组件
[bugfix](auth): 修复登录表单验证问题
[performance](tree): 优化族谱渲染性能
[refactor](components): 重构公共组件结构
[format]: 代码格式化调整
[docs]: 更新组件文档
[test](family): 添加族谱组件单元测试
[chore]: 更新依赖包版本
```
